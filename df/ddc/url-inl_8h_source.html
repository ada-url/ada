37  ada_log("parse_port('", view, "') ", view.size());
38  uint16_t parsed_port{};
39  auto r = std::from_chars(view.data(), view.data() + view.size(), parsed_port);
40  if(r.ec == std::errc::result_out_of_range) {
41  ada_log("parse_port: std::errc::result_out_of_range");
