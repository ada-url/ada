99 ada_really_inline uint64_t swap_bytes(uint64_t val) noexcept {
100 // performance: this often compiles to a single instruction (e.g., bswap)
101 return ((((val) & 0xff00000000000000ull) >> 56) |
102 (((val) & 0x00ff000000000000ull) >> 40) |
103 (((val) & 0x0000ff0000000000ull) >> 24) |
104 (((val) & 0x000000ff00000000ull) >> 8 ) |
105 (((val) & 0x00000000ff000000ull) << 8 ) |
106 (((val) & 0x0000000000ff0000ull) << 24) |

126 // you *really* want find_next_host_delimiter to be inlined, because
127 // otherwise, the constants may get reloaded each time (bad).
128 auto has_zero_byte = [](uint64_t v) {
129 return ((v - 0x0101010101010101) & ~(v)&0x8080808080808080);
130 };
131 auto index_of_first_set_byte = [](uint64_t v) {
132 return ((((v - 1) & 0x101010101010101) * 0x101010101010101) >> 56) - 1;
133 };
134 auto broadcast = [](uint8_t v) -> uint64_t { return 0x101010101010101 * v; };
135 size_t i = location;
136 uint64_t mask1 = broadcast(':');
137 uint64_t mask2 = broadcast('/');
138 uint64_t mask3 = broadcast('\\');
139 uint64_t mask4 = broadcast('?');
140 uint64_t mask5 = broadcast('[');
141 // This loop will get autovectorized under many optimizing compilers,
142 // so you get actually SIMD!
143 for (; i + 7 < view.size(); i += 8) {
144 uint64_t word{};
145 // performance: the next memcpy translates into a single CPU instruction.
146 memcpy(&word, view.data() + i, sizeof(word));
147 // performance: on little-endian systems (most systems), this next line is free.
148 word = swap_bytes_if_big_endian(word);
149 uint64_t xor1 = word ^ mask1;
150 uint64_t xor2 = word ^ mask2;
151 uint64_t xor3 = word ^ mask3;
152 uint64_t xor4 = word ^ mask4;
153 uint64_t xor5 = word ^ mask5;
154 uint64_t is_match = has_zero_byte(xor1) | has_zero_byte(xor2) | has_zero_byte(xor3) | has_zero_byte(xor4) | has_zero_byte(xor5);
155 if(is_match) {
156 return i + index_of_first_set_byte(is_match);
157 }
158 }
159 if (i < view.size()) {
160 uint64_t word{};
161 // performance: the next memcpy translates into a function call, but
162 // that is difficult to avoid. Might be a bit expensive.
163 memcpy(&word, view.data() + i, view.size() - i);
164 word = swap_bytes_if_big_endian(word);
165 uint64_t xor1 = word ^ mask1;

243 const size_t view_size = view.size();
244 size_t location = 0;
245 bool found_colon = false;
264 if(is_special) {
265 // We move to the next delimiter.
266 location = find_next_host_delimiter_special(view, location);
267 // Unless we find '[' then we are going only going to have to call
268 // find_next_host_delimiter_special once.
269 for (;location < view_size; location = find_next_host_delimiter_special(view, location)) {
270 if (view[location] == '[') {

272 if (location == std::string_view::npos) {
273 // performance: view.find might get translated to a memchr, which
274 // has no notion of std::string_view::npos, so the code does not
275 // reflect the assembly.

277 break;
278 }
279 } else {
280 found_colon = view[location] == ':';
281 break;
282 }
283 }
284 } else {
285 // We move to the next delimiter.
286 location = find_next_host_delimiter(view, location);
287 // Unless we find '[' then we are going only going to have to call
288 // find_next_host_delimiter_special once.
289 for (;location < view_size; location = find_next_host_delimiter(view, location)) {
290 if (view[location] == '[') {

316 ada_really_inline bool parse_prepared_path(std::string_view input, ada::scheme::type type, std::string& path) {
317 ada_log("parse_path ", input);
318 uint8_t accumulator = checkers::path_signature(input);
319 // Let us first detect a trivial case.
320 // If it is special, we check that we have no dot, no %, no \ and no
321 // character needing percent encoding. Otherwise, we check that we have no %,
322 // no dot, and no character needing percent encoding.
323 bool special = type != ada::scheme::NOT_SPECIAL;
324 bool trivial_path =
325 (special ? (accumulator == 0) : ((accumulator & 0b11111101) == 0)) &&
326 (type != ada::scheme::type::FILE);
327 if(accumulator == 4) {
328 // '4' means that we have at least one dot, but nothing that requires
329 // percent encoding or decoding. The only part that is not trivial is
330 // that we may have single dots and double dots path segments.
331 // If we have such segments, then we either have a path that begins
332 // with '.' (easy to check), or we have the sequence './'.
333 // Note: input cannot be empty, it must at least contain one character ('.')
334 // Note: we know that '\' is not present.
335 if(input[0] != '.' && input.find("/.") == std::string_view::npos) { trivial_path = true; }
336 }
337 if (trivial_path) {
338 ada_log("parse_path trivial");
339 path += '/';
340 path += input;
341 return true;
342 }
343 // We are going to need to look a bit at the path, but let us see if we can
344 // ignore percent encoding *and* backslashes *and* percent characters.
345 // Except for the trivial case, this is likely to capture 99% of paths out
346 // there.
347 bool fast_path = (special && (accumulator & 0b11111011) == 0) &&
348 (type != ada::scheme::type::FILE);
349 if (fast_path) {
350 ada_log("parse_path fast");
351 // Here we don't need to worry about \ or percent encoding.
352 // We also do not have a file protocol. We might have dots, however,
353 // but dots must as appear as '.', and they cannot be encoded because
354 // the symbol '%' is not present.
355 size_t previous_location = 0; // We start at 0.
356 do {
357 size_t new_location = input.find('/', previous_location);
358 //std::string_view path_view = input;
359 // We process the last segment separately:
360 if (new_location == std::string_view::npos) {
361 std::string_view path_view = input.substr(previous_location);
362 if (path_view == "..") { // The path ends with ..
363 // e.g., if you receive ".." with an empty path, you go to "/".
364 if(path.empty()) { path = '/'; return true; }
365 // Fast case where we have nothing to do:
366 if(path.back() == '/') { return true; }
367 // If you have the path "/joe/myfriend",
368 // then you delete 'myfriend'.
369 path.resize(path.rfind('/') + 1);
370 return true;
371 }
372 path += '/';
373 if (path_view != ".") {
374 path.append(path_view);
375 }
376 return true;
377 } else {
378 // This is a non-final segment.
379 std::string_view path_view = input.substr(previous_location, new_location - previous_location);
380 previous_location = new_location + 1;
381 if (path_view == "..") {
382 if(!path.empty()) { path.erase(path.rfind('/')); }
383 } else if (path_view != ".") {
384 path += '/';
385 path.append(path_view);
386 }
387 }
388 } while (true);
389 } else {
390 ada_log("parse_path slow");
391 // we have reached the general case
392 bool needs_percent_encoding = (accumulator & 1);
393 std::string path_buffer_tmp;
394 do {
395 size_t location = (special && (accumulator & 2))
396 ? input.find_first_of("/\\")
397 : input.find('/');
398 std::string_view path_view = input;
399 if (location != std::string_view::npos) {
400 path_view.remove_suffix(path_view.size() - location);
401 input.remove_prefix(location + 1);
402 }
403 // path_buffer is either path_view or it might point at a percent encoded temporary file.
404 std::string_view path_buffer =
405 (needs_percent_encoding
406 && ada::unicode::percent_encode(path_view, character_sets::PATH_PERCENT_ENCODE, path_buffer_tmp)) ?
407 path_buffer_tmp :
408 path_view;
409 if (unicode::is_double_dot_path_segment(path_buffer)) {
410 helpers::shorten_path(path, type);
411 if (location == std::string_view::npos) {
412 path += '/';
413 }
414 } else if (unicode::is_single_dot_path_segment(path_buffer) &&
415 (location == std::string_view::npos)) {
416 path += '/';
417 }
418 // Otherwise, if path_buffer is not a single-dot path segment, then:
419 else if (!unicode::is_single_dot_path_segment(path_buffer)) {
420 // If url’s scheme is "file", url’s path is empty, and path_buffer is a
421 // Windows drive letter, then replace the second code point in
422 // path_buffer with U+003A (:).
423 if (type == ada::scheme::type::FILE && path.empty() &&
424 checkers::is_windows_drive_letter(path_buffer)) {
425 path += '/';
426 path += path_buffer[0];
427 path += ':';
428 path_buffer.remove_prefix(2);
429 path.append(path_buffer);
430 } else {
431 // Append path_buffer to url’s path.
432 path += '/';
433 path.append(path_buffer);
434 }
435 }
436 if (location == std::string_view::npos) {
437 return true;
438 }
439 } while (true);
440 }
441 }
