126 ada_really_inline bool parse_prepared_path(std::string_view input, ada::scheme::type type, std::string& path) {
127 ada_log("parse_path ", input);
128 uint8_t accumulator = checkers::path_signature(input);
129 // Let us first detect a trivial case.
130 // If it is special, we check that we have no dot, no %, no \ and no
131 // character needing percent encoding. Otherwise, we check that we have no %,
132 // no dot, and no character needing percent encoding.
133 bool special = type != ada::scheme::NOT_SPECIAL;
134 bool trivial_path =
135 (special ? (accumulator == 0) : ((accumulator & 0b11111101) == 0)) &&
136 (type != ada::scheme::type::FILE);
137 if (trivial_path) {
138 ada_log("parse_path trivial");
139 path += '/';
140 path += input;
141 return true;
142 }
143 // We are going to need to look a bit at the path, but let us see if we can
144 // ignore percent encoding *and* backslashes *and* percent characters.
145 // Except for the trivial case, this is likely to capture 99% of paths out
146 // there.
147 bool fast_path = (special && (accumulator & 0b11111011) == 0) &&
148 (type != ada::scheme::type::FILE);
149 if (fast_path) {
150 ada_log("parse_path fast");
151 // Here we don't need to worry about \ or percent encoding.
152 // We also do not have a file protocol. We might have dots, however,
153 // but dots must as appear as '.', and they cannot be encoded because
154 // the symbol '%' is not present.
155 size_t previous_location = 0; // We start at 0.
156 do {
157 size_t new_location = input.find('/', previous_location);
158 //std::string_view path_view = input;
159 // We process the last segment separately:
160 if (new_location == std::string_view::npos) {
161 std::string_view path_view = input.substr(previous_location);
162 if (path_view == "..") { // The path ends with ..
163 // e.g., if you receive ".." with an empty path, you go to "/".
164 if(path.empty()) { path = '/'; return true; }
165 // Fast case where we have nothing to do:
166 if(path.back() == '/') { return true; }
167 // If you have the path "/joe/myfriend",
168 // then you delete 'myfriend'.
169 path.resize(path.rfind('/') + 1);
170 return true;
171 }
172 path += '/';
173 if (path_view != ".") {
174 path.append(path_view);
175 }
176 return true;
177 } else {
178 // This is a non-final segment.
179 std::string_view path_view = input.substr(previous_location, new_location - previous_location);
180 previous_location = new_location + 1;
181 if (path_view == "..") {
182 if(!path.empty()) { path.erase(path.rfind('/')); }
183 } else if (path_view != ".") {
184 path += '/';
185 path.append(path_view);
186 }
187 }
188 } while (true);
189 } else {
190 ada_log("parse_path slow");
191 // we have reached the general case
192 bool needs_percent_encoding = (accumulator & 1);
193 std::string path_buffer_tmp;
194 do {
195 size_t location = (special && (accumulator & 2))
196 ? input.find_first_of("/\\")
197 : input.find('/');
198 std::string_view path_view = input;
199 if (location != std::string_view::npos) {
200 path_view.remove_suffix(path_view.size() - location);
201 input.remove_prefix(location + 1);
202 }
203 // path_buffer is either path_view or it might point at a percent encoded temporary file.
204 std::string_view path_buffer =
205 (needs_percent_encoding
206 && ada::unicode::percent_encode(path_view, character_sets::PATH_PERCENT_ENCODE, path_buffer_tmp)) ?
207 path_buffer_tmp :
208 path_view;
209 if (unicode::is_double_dot_path_segment(path_buffer)) {
210 helpers::shorten_path(path, type);
211 if (location == std::string_view::npos) {
212 path += '/';
213 }
214 } else if (unicode::is_single_dot_path_segment(path_buffer) &&
215 (location == std::string_view::npos)) {
216 path += '/';
217 }
218 // Otherwise, if path_buffer is not a single-dot path segment, then:
219 else if (!unicode::is_single_dot_path_segment(path_buffer)) {
220 // If url’s scheme is "file", url’s path is empty, and path_buffer is a
221 // Windows drive letter, then replace the second code point in
222 // path_buffer with U+003A (:).
223 if (type == ada::scheme::type::FILE && path.empty() &&
224 checkers::is_windows_drive_letter(path_buffer)) {
225 path += '/';
226 path += path_buffer[0];
227 path += ':';
228 path_buffer.remove_prefix(2);
229 path.append(path_buffer);
230 } else {
231 // Append path_buffer to url’s path.
232 path += '/';
233 path.append(path_buffer);
234 }
235 }
236 if (location == std::string_view::npos) {
237 return true;
238 }
239 } while (true);
240 }
