517  // Note: it is unsafe to do base_url->scheme unless you know that
518  // base_url_has_value() is true.
519  if (base_url != nullptr && base_url->get_scheme_type() == ada::scheme::type::FILE) {
520  // Set url’s host to base’s host.
521  url.host = base_url->host;
522 
523  // If the code point substring from pointer to the end of input does not start with
524  // a Windows drive letter and base’s path[0] is a normalized Windows drive letter,
525  // then append base’s path[0] to url’s path.
526  if (!base_url->path.empty()) {
527  if (!checkers::is_windows_drive_letter(helpers::substring(url_data, input_position))) {
528  std::string_view first_base_url_path = base_url->path;
529  first_base_url_path.remove_prefix(1);
530  size_t loc = first_base_url_path.find('/');
531  if(loc != std::string_view::npos) {
532  first_base_url_path.remove_suffix(first_base_url_path.size() - loc);
533  }
534  if (checkers::is_normalized_windows_drive_letter(first_base_url_path)) {
535  url.path += '/';
536  url.path += first_base_url_path;
537  }
538  }
539  }
540  }
541 
542  // Set state to path state, and decrease pointer by 1.
543  state = ada::state::PATH;
544  }
545 
546  break;
547  }
548  case ada::state::FILE_HOST: {
549  std::string_view view = helpers::substring(url_data, input_position);
550  ada_log("FILE_HOST ", helpers::substring(url_data, input_position));
551 
552  size_t location = view.find_first_of("/\\?");
553  std::string_view file_host_buffer(view.data(), (location != std::string_view::npos) ? location : view.size());
554 
555  if (checkers::is_windows_drive_letter(file_host_buffer)) {
556  state = ada::state::PATH;
557  } else if (file_host_buffer.empty()) {
558  // Set url’s host to the empty string.
559  url.host = "";
560  // Set state to path start state.
561  state = ada::state::PATH_START;
562  } else {
563  size_t consumed_bytes = file_host_buffer.size();
564  input_position += consumed_bytes;
565  // Let host be the result of host parsing buffer with url is not special.
566  if(!url.parse_host(file_host_buffer)) { return url; }
567 
568  // If host is "localhost", then set host to the empty string.
569  if (url.host.has_value() && url.host.value() == "localhost") {
570  url.host = "";
571  }
572 
573  // Set buffer to the empty string and state to path start state.
574  state = ada::state::PATH_START;
575  }
576 
577  break;
578  }
579  case ada::state::FILE: {
580  ada_log("FILE ", helpers::substring(url_data, input_position));
581  std::string_view file_view = helpers::substring(url_data, input_position);
582 
583  // Set url’s scheme to "file".
584  url.set_scheme("file");
585 
586  // Set url’s host to the empty string.
587  url.host = "";
588 
589  // If c is U+002F (/) or U+005C (\‍), then:
590  if (input_position != input_size && (url_data[input_position] == '/' || url_data[input_position] == '\\')) {
591  ada_log("FILE c is U+002F or U+005C");
592  // Set state to file slash state.
593  state = ada::state::FILE_SLASH;
594  }
595  // Otherwise, if base is non-null and base’s scheme is "file":
596  else if (base_url != nullptr && base_url->get_scheme_type() == ada::scheme::type::FILE) {
597  // Set url’s host to base’s host, url’s path to a clone of base’s path, and url’s query to base’s query.
598  ada_log("FILE base non-null");
599  url.host = base_url->host;
600  url.path = base_url->path;
601  url.has_opaque_path = base_url->has_opaque_path;
602  url.query = base_url->query;
603 
604  // If c is U+003F (?), then set url’s query to the empty string and state to query state.
605  if (input_position != input_size && url_data[input_position] == '?') {
606  state = ada::state::QUERY;
607  }
608  // Otherwise, if c is not the EOF code point:
609  else if (input_position != input_size) {
610  // Set url’s query to null.
611  url.query = std::nullopt;
612 
613  // If the code point substring from pointer to the end of input does not start with a
614  // Windows drive letter, then shorten url’s path.
615  if (!checkers::is_windows_drive_letter(file_view)) {
616  helpers::shorten_path(url.path, url.get_scheme_type());
617  }
618  // Otherwise:
619  else {
620  // Set url’s path to an empty list.
621  url.path.clear();
622  url.has_opaque_path = true;
623  }
624 
625  // Set state to path state and decrease pointer by 1.
626  state = ada::state::PATH;
627  break;
628  }
629  }
630  // Otherwise, set state to path state, and decrease pointer by 1.
631  else {
632  ada_log("FILE go to path");
633  state = ada::state::PATH;
634  break;
635  }
636 
637  input_position++;
638  break;
639  }
640  default:
641  ada::unreachable();
642  }
643  }
644  ada_log("returning ", url.to_string());
645  return url;
646  }
