545 std::string_view file_host_buffer(view.data(), (location != std::string_view::npos) ? location : view.size());
546
547 if (checkers::is_windows_drive_letter(file_host_buffer)) {
548 state = ada::state::PATH;
549 } else if (file_host_buffer.empty()) {
550 // Set url’s host to the empty string.
551 url.host = "";
552 // Set state to path start state.
553 state = ada::state::PATH_START;
554 } else {
555 size_t consumed_bytes = file_host_buffer.size();
556 input_position += consumed_bytes;
557 // Let host be the result of host parsing buffer with url is not special.
558 if(!url.parse_host(file_host_buffer)) { return url; }
559
560 // If host is "localhost", then set host to the empty string.
561 if (url.host.has_value() && url.host.value() == "localhost") {
562 url.host = "";
563 }
564
565 // Set buffer to the empty string and state to path start state.
566 state = ada::state::PATH_START;
567 }
568
569 break;
570 }
571 case ada::state::FILE: {
572 ada_log("FILE ", helpers::substring(url_data, input_position));
573 std::string_view file_view = helpers::substring(url_data, input_position);
574
575 // Set url’s scheme to "file".
576 url.set_scheme("file");
577
578 // Set url’s host to the empty string.
579 url.host = "";
580
581 // If c is U+002F (/) or U+005C (\‍), then:
582 if (input_position != input_size && (url_data[input_position] == '/' || url_data[input_position] == '\\')) {
583 ada_log("FILE c is U+002F or U+005C");
584 // Set state to file slash state.
585 state = ada::state::FILE_SLASH;
586 }
587 // Otherwise, if base is non-null and base’s scheme is "file":
588 else if (base_url != nullptr && base_url->get_scheme_type() == ada::scheme::type::FILE) {
589 // Set url’s host to base’s host, url’s path to a clone of base’s path, and url’s query to base’s query.
590 ada_log("FILE base non-null");
591 url.host = base_url->host;
592 url.path = base_url->path;
593 url.has_opaque_path = base_url->has_opaque_path;
594 url.query = base_url->query;
595
596 // If c is U+003F (?), then set url’s query to the empty string and state to query state.
597 if (input_position != input_size && url_data[input_position] == '?') {
598 state = ada::state::QUERY;
599 }
600 // Otherwise, if c is not the EOF code point:
601 else if (input_position != input_size) {
602 // Set url’s query to null.
603 url.query = std::nullopt;
604
605 // If the code point substring from pointer to the end of input does not start with a
606 // Windows drive letter, then shorten url’s path.
607 if (!checkers::is_windows_drive_letter(file_view)) {
608 helpers::shorten_path(url.path, url.get_scheme_type());
609 }
610 // Otherwise:
611 else {
612 // Set url’s path to an empty list.
613 url.path.clear();
614 url.has_opaque_path = true;
615 }
616
617 // Set state to path state and decrease pointer by 1.
618 state = ada::state::PATH;
619 break;
620 }
621 }
622 // Otherwise, set state to path state, and decrease pointer by 1.
623 else {
624 ada_log("FILE go to path");
625 state = ada::state::PATH;
626 break;
627 }
628
629 input_position++;
630 break;
631 }
632 default:
633 ada::unreachable();
634 }
635 }
636 ada_log("returning ", url.to_string());
637 return url;
638 }
