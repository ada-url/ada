167  bool url::parse_ipv4(std::string_view input) {
168  ada_log("parse_ipv4 ", input, "[", input.size(), " bytes]");
169  if(input.back()=='.') {
170  input.remove_suffix(1);
171  }
172  size_t digit_count{0};
173  uint64_t ipv4{0};
174  // we could unroll for better performance?
175  for(;(digit_count < 4) && !(input.empty()); digit_count++) {
176  uint32_t result{}; // If any number exceeds 32 bits, we have an error.
177  bool is_hex = checkers::has_hex_prefix(input);
178  if(is_hex && ((input.length() == 2)|| ((input.length() > 2) && (input[2]=='.')))) {
179  // special case
180  result = 0;
181  input.remove_prefix(2);
182  } else {
183  std::from_chars_result r;
184  if(is_hex) {
185  r = std::from_chars(input.data() + 2, input.data() + input.size(), result, 16);
186  } else if ((input.length() >= 2) && input[0] == '0' && checkers::is_digit(input[1])) {
187  r = std::from_chars(input.data() + 1, input.data() + input.size(), result, 8);
188  } else {
189  r = std::from_chars(input.data(), input.data() + input.size(), result, 10);
190  }
191  if (r.ec != std::errc()) { return is_valid = false; }
192  input.remove_prefix(r.ptr-input.data());
193  }
194  if(input.empty()) {
195  // We have the last value.
196  // At this stage, ipv4 contains digit_count*8 bits.
197  // So we have 32-digit_count*8 bits left.
198  if(result > (uint64_t(1)<<(32-digit_count*8))) { return is_valid = false; }
199  ipv4 <<=(32-digit_count*8);
200  ipv4 |= result;
201  goto final;
202  } else {
203  // There is more, so that the value must no be larger than 255
204  // and we must have a '.'.
205  if ((result>255) || (input[0]!='.')) { return is_valid = false; }
206  ipv4 <<=8;
207  ipv4 |= result;
208  input.remove_prefix(1); // remove '.'
209  }
210  }
211  if((digit_count != 4) || (!input.empty())) {return is_valid = false; }
212  final:
213  // We could also check result.ptr to see where the parsing ended.
214  host = ada::serializers::ipv4(ipv4);
215  return true;
