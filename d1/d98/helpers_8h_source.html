191inline int leading_zeroes(uint32_t input_num) noexcept {
192#if ADA_REGULAR_VISUAL_STUDIO
193 unsigned long leading_zero(0);
194 unsigned long in(input_num);
195 return _BitScanReverse(&leading_zero, in) ? int(31 - leading_zero) : 32;
196#else
197 return __builtin_clz(input_num);
198#endif // ADA_REGULAR_VISUAL_STUDIO
199}

206inline int fast_digit_count(uint32_t x) noexcept {
207 auto int_log2 = [](uint32_t z) -> int {
208 return 31 - ada::helpers::leading_zeroes(z | 1);
209 };
210 // Compiles to very few instructions. Note that the
211 // table is static and thus effectively a constant.
212 // We leave it inside the function because it is meaningless
213 // outside of it (this comes at no performance cost).
214 const static uint64_t table[] = {
215 4294967296, 8589934582, 8589934582, 8589934582, 12884901788,
216 12884901788, 12884901788, 17179868184, 17179868184, 17179868184,
217 21474826480, 21474826480, 21474826480, 21474826480, 25769703776,
218 25769703776, 25769703776, 30063771072, 30063771072, 30063771072,
219 34349738368, 34349738368, 34349738368, 34349738368, 38554705664,
220 38554705664, 38554705664, 41949672960, 41949672960, 41949672960,
221 42949672960, 42949672960};
222 return int((x + table[int_log2(x)]) >> 32);
223}
