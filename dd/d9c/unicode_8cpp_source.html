23constexpr bool to_lower_ascii(char* input, size_t length) noexcept {
24 uint64_t broadcast_80 = broadcast(0x80);
25 uint64_t broadcast_Ap = broadcast(128 - 'A');
26 uint64_t broadcast_Zp = broadcast(128 - 'Z' - 1);
27 uint64_t non_ascii = 0;
28 size_t i = 0;
29
30 for (; i + 7 < length; i += 8) {
31 uint64_t word{};
32 memcpy(&word, input + i, sizeof(word));
33 non_ascii |= (word & broadcast_80);
34 word ^=
35 (((word + broadcast_Ap) ^ (word + broadcast_Zp)) & broadcast_80) >> 2;
36 memcpy(input + i, &word, sizeof(word));
37 }
38 if (i < length) {
39 uint64_t word{};
40 memcpy(&word, input + i, length - i);
41 non_ascii |= (word & broadcast_80);
42 word ^=
43 (((word + broadcast_Ap) ^ (word + broadcast_Zp)) & broadcast_80) >> 2;
44 memcpy(input + i, &word, length - i);
45 }
46 return non_ascii == 0;
47}

49ada_really_inline bool has_tabs_or_newline(
50 std::string_view user_input) noexcept {
51 // first check for short strings in which case we do it naively.
52 if (user_input.size() < 16) { // slow path
53 for (size_t i = 0; i < user_input.size(); i++) {
54 if (user_input[i] == '\r' || user_input[i] == '\n' ||
55 user_input[i] == '\t') {
56 return true;
57 }
58 }
59 return false;
60 }
61 // fast path for long strings (expected to be common)
62 size_t i = 0;
63 const uint8x16_t mask1 = vmovq_n_u8('\r');
64 const uint8x16_t mask2 = vmovq_n_u8('\n');
65 const uint8x16_t mask3 = vmovq_n_u8('\t');
66 uint8x16_t running{0};
67 for (; i + 15 < user_input.size(); i += 16) {
68 uint8x16_t word = vld1q_u8((const uint8_t*)user_input.data() + i);
69 running = vorrq_u8(vorrq_u8(running, vorrq_u8(vceqq_u8(word, mask1),
70 vceqq_u8(word, mask2))),
71 vceqq_u8(word, mask3));
72 }
73 if (i < user_input.size()) {
74 uint8x16_t word =
75 vld1q_u8((const uint8_t*)user_input.data() + user_input.length() - 16);
76 running = vorrq_u8(vorrq_u8(running, vorrq_u8(vceqq_u8(word, mask1),
77 vceqq_u8(word, mask2))),
78 vceqq_u8(word, mask3));
79 }
80 return vmaxvq_u8(running) != 0;

119ada_really_inline bool has_tabs_or_newline(
120 std::string_view user_input) noexcept {
121 auto has_zero_byte = [](uint64_t v) {
122 return ((v - 0x0101010101010101) & ~(v)&0x8080808080808080);
123 };
124 size_t i = 0;
125 uint64_t mask1 = broadcast('\r');
126 uint64_t mask2 = broadcast('\n');
127 uint64_t mask3 = broadcast('\t');
128 uint64_t running{0};

130 uint64_t word{};
131 memcpy(&word, user_input.data() + i, sizeof(word));
132 uint64_t xor1 = word ^ mask1;
133 uint64_t xor2 = word ^ mask2;
134 uint64_t xor3 = word ^ mask3;
135 running |= has_zero_byte(xor1) | has_zero_byte(xor2) | has_zero_byte(xor3);
136 }

142 uint64_t xor3 = word ^ mask3;
143 running |= has_zero_byte(xor1) | has_zero_byte(xor2) | has_zero_byte(xor3);
144 }

289ada_really_inline ada_constexpr bool is_double_dot_path_segment(
290 std::string_view input) noexcept {
291 // This will catch most cases:
292 // The length must be 2,4 or 6.
293 // We divide by two and require
294 // that the result be between 1 and 3 inclusively.
295 uint64_t half_length = uint64_t(input.size()) / 2;
296 if (half_length - 1 > 2) {
297 return false;
298 }
299 // We have a string of length 2, 4 or 6.
300 // We now check the first character:
301 if ((input[0] != '.') && (input[0] != '%')) {
302 return false;
303 }
304 // We are unlikely the get beyond this point.
305 int hash_value = (input.size() + (unsigned)(input[0])) & 3;
306 const std::string_view target = table_is_double_dot_path_segment[hash_value];
307 if (target.size() != input.size()) {
308 return false;
309 }
310 // We almost never get here.
311 // Optimizing the rest is relatively unimportant.
312 auto prefix_equal_unsafe = [](std::string_view a, std::string_view b) {
313 uint16_t A, B;
314 memcpy(&A, a.data(), sizeof(A));
315 memcpy(&B, b.data(), sizeof(B));
316 return A == B;
317 };
318 if (!prefix_equal_unsafe(input, target)) {
319 return false;
320 }
321 for (size_t i = 2; i < input.size(); i++) {
322 char c = input[i];
323 if ((uint8_t((c | 0x20) - 0x61) <= 25 ? (c | 0x20) : c) != target[i]) {
324 return false;
325 }
326 }
327 return true;
328 // The above code might be a bit better than the code below. Compilers
329 // are not stupid and may use the fact that these strings have length 2,4 and
330 // 6 and other tricks.
331 // return input == ".." ||
332 // input == ".%2e" || input == ".%2E" ||
333 // input == "%2e." || input == "%2E." ||
334 // input == "%2e%2e" || input == "%2E%2E" || input == "%2E%2e" || input ==
335 // "%2e%2E";
336}

355std::string percent_decode(const std::string_view input, size_t first_percent) {
356 // next line is for safety only, we expect users to avoid calling
357 // percent_decode when first_percent is outside the range.
358 if (first_percent == std::string_view::npos) {
359 return std::string(input);
360 }
361 std::string dest;
362 dest.reserve(input.length());
363 dest.append(input.substr(0, first_percent));
364 const char* pointer = input.data() + first_percent;
365 const char* end = input.data() + input.size();
366 // Optimization opportunity: if the following code gets
367 // called often, it can be optimized quite a bit.
368 while (pointer < end) {
369 const char ch = pointer[0];
370 size_t remaining = end - pointer - 1;
371 if (ch != '%' || remaining < 2 ||
372 ( // ch == '%' && // It is unnecessary to check that ch == '%'.
373 (!is_ascii_hex_digit(pointer[1]) ||
374 !is_ascii_hex_digit(pointer[2])))) {
375 dest += ch;
376 pointer++;
377 continue;
378 } else {
379 unsigned a = convert_hex_to_binary(pointer[1]);
380 unsigned b = convert_hex_to_binary(pointer[2]);
381 char c = static_cast<char>(a * 16 + b);
382 dest += c;
383 pointer += 3;
384 }
385 }
386 return dest;
387}

389std::string percent_encode(const std::string_view input,
390 const uint8_t character_set[]) {
391 auto pointer =
392 std::find_if(input.begin(), input.end(), [character_set](const char c) {
393 return character_sets::bit_at(character_set, c);
394 });
395 // Optimization: Don't iterate if percent encode is not required
396 if (pointer == input.end()) {
397 return std::string(input);
398 }
399
400 std::string result;
401 result.reserve(input.length()); // in the worst case, percent encoding might
402 // produce 3 characters.
403 result.append(input.substr(0, std::distance(input.begin(), pointer)));
404
405 for (; pointer != input.end(); pointer++) {
406 if (character_sets::bit_at(character_set, *pointer)) {
407 result.append(character_sets::hex + uint8_t(*pointer) * 4, 3);
408 } else {
409 result += *pointer;
410 }
411 }
412
413 return result;
414}

451bool to_ascii(std::optional<std::string>& out, const std::string_view plain,
452 size_t first_percent) {
453 std::string percent_decoded_buffer;
454 std::string_view input = plain;
455 if (first_percent != std::string_view::npos) {
456 percent_decoded_buffer = unicode::percent_decode(plain, first_percent);
457 input = percent_decoded_buffer;
458 }

461 if (idna_ascii.empty() || contains_forbidden_domain_code_point(
462 idna_ascii.data(), idna_ascii.size())) {
463 return false;
464 }
465 out = std::move(idna_ascii);
