438  }

441  if (leading_byte < 0b10000000) {

443  *utf16_output++ = char16_t(leading_byte);
444  pos++;
445  } else if ((leading_byte & 0b11100000) == 0b11000000) {
446  // We have a two-byte UTF-8, it should become
447  // a single UTF-16 word.
448  if(pos + 1 >= len) { return 0; } // minimal bound checking
449  if ((data[pos + 1] & 0b11000000) != 0b10000000) { return 0; }
450  // range check
451  uint32_t code_point = (leading_byte & 0b00011111) << 6 | (data[pos + 1] & 0b00111111);
452  if (code_point < 0x80 || 0x7ff < code_point) { return 0; }
453  *utf16_output++ = char16_t(code_point);
454  pos += 2;
455  } else if ((leading_byte & 0b11110000) == 0b11100000) {
456  // We have a three-byte UTF-8, it should become
457  // a single UTF-16 word.
458  if(pos + 2 >= len) { return 0; } // minimal bound checking
459 
460  if ((data[pos + 1] & 0b11000000) != 0b10000000) { return 0; }
461  if ((data[pos + 2] & 0b11000000) != 0b10000000) { return 0; }
462  // range check
463  uint32_t code_point = (leading_byte & 0b00001111) << 12 |
464  (data[pos + 1] & 0b00111111) << 6 |
465  (data[pos + 2] & 0b00111111);
466  if (code_point < 0x800 || 0xffff < code_point ||
467  (0xd7ff < code_point && code_point < 0xe000)) {
468  return 0;
469  }
470  *utf16_output++ = char16_t(code_point);
471  pos += 3;
472  } else if ((leading_byte & 0b11111000) == 0b11110000) { // 0b11110000
473  // we have a 4-byte UTF-8 word.
474  if(pos + 3 >= len) { return 0; } // minimal bound checking
475  if ((data[pos + 1] & 0b11000000) != 0b10000000) { return 0; }
476  if ((data[pos + 2] & 0b11000000) != 0b10000000) { return 0; }
477  if ((data[pos + 3] & 0b11000000) != 0b10000000) { return 0; }
478 
479  // range check
480  uint32_t code_point =
481  (leading_byte & 0b00000111) << 18 | (data[pos + 1] & 0b00111111) << 12 |
482  (data[pos + 2] & 0b00111111) << 6 | (data[pos + 3] & 0b00111111);
483  if (code_point <= 0xffff || 0x10ffff < code_point) { return 0; }
484  code_point -= 0x10000;
485  uint16_t high_surrogate = uint16_t(0xD800 + (code_point >> 10));
486  uint16_t low_surrogate = uint16_t(0xDC00 + (code_point & 0x3FF));
487  *utf16_output++ = char16_t(high_surrogate);
488  *utf16_output++ = char16_t(low_surrogate);
489  pos += 4;
490  } else {
491  return 0;
492  }
493  }
494  return int(utf16_output - start);
495  };

498  out = std::nullopt;
499  return false;
500  }

502

506  }

510

512  if(required_buffer_size == 0) {
513  out = std::nullopt;
514  return false;
515  }
516  // This will not validate the punycode, so let us work it in reverse.

523  for(size_t i = 0; i < required_buffer_size; i++) { (*out)[i] = char(ascii_buffer.get()[i]); }
524 #else
