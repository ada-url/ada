82ada_owned_string ada_get_origin(ada_url result) noexcept {
83 ada::result<ada::url_aggregator>& r = get_instance(result);
84 ada_owned_string owned;
85 if (!r) {
86 owned.data = nullptr;
87 owned.length = 0;
88 return owned;
89 }
90 std::string out = r->get_origin();
91 owned.length = out.size();
92 owned.data = new char[owned.length];
93 memcpy((void*)owned.data, out.data(), owned.length);
94 return owned;
95}
